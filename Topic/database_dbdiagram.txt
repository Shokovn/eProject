Table User {
  user_id int [pk, increment]
  fullname varchar
  username varchar
  email varchar
  phone_number varchar
  password varchar
  role varchar
  created_at datetime
  updated_at datetime
}

Table Category {
  category_id int [pk, increment]
  category_name varchar
  category_thumbnail varchar
}

Table Brand {
  brand_id int [pk, increment]
  brand_name varchar
  brand_thumbnail varchar
}

Table Product {
  product_id int [pk, increment]
  product_title varchar
  product_description text
  product_price decimal
  product_content text
  product_quantity int
  product_thumbnail varchar

  product_img_id int
  category_id int
  brand_id int

  created_at datetime
  updated_at datetime
}

Table Product_img {
  product_img_id int [pk, increment]
  product_id int
  url varchar
}

Table Gallery {
  img_id int [pk, increment]
  url varchar
}

Table Cart {
  cart_id int [pk, increment]
  user_id int
  product_id int
  quantity int
  created_at datetime
}

Table Content {
  content_id int [pk, increment]
  sale varchar
  product_id int
  banner_id int
  banner_img varchar
}

Table Feedback {
  feedback_id int [pk, increment]
  order_id int
  user_id int
  product_id int
  feedback_content text
  rating int
  is_public boolean
  created_at datetime
}

Table Order {
  order_id int [pk, increment]
  user_id int
  feedback_id int
  order_status varchar
  total_amount decimal
  receiver varchar
  address varchar
  order_date datetime
}

Table Order_detail {
  order_id int
  product_id int
  quantity int
  unit_price decimal
  created_at datetime

  Indexes {
    (order_id, product_id) [pk]
  }
}
